create or replace function set_updated_at()
returns trigger
language plpgsql
as $$
begin
    new.updated_at = now();
    return new;
end;
$$;

create table if not exists accounts (
    id uuid primary key default gen_random_uuid(),
    name text not null,
    slug text not null unique,
    status text not null default 'active' check (status in ('active', 'paused', 'archived')),
    created_at timestamptz not null default now(),
    updated_at timestamptz not null default now()
);

create table if not exists bot_settings (
    id uuid primary key default gen_random_uuid(),
    account_id uuid not null unique references accounts(id) on delete cascade,
    timezone text not null default 'Europe/Belgrade',
    currency text not null default 'RSD',
    locale text not null default 'ru',
    shift_tracking_enabled boolean not null default true,
    zones_enabled boolean not null default true,
    service_types_enabled boolean not null default false,
    created_at timestamptz not null default now(),
    updated_at timestamptz not null default now()
);

create table if not exists zones (
    id uuid primary key default gen_random_uuid(),
    account_id uuid not null references accounts(id) on delete cascade,
    name text not null,
    sort_order integer not null default 0,
    is_active boolean not null default true,
    created_at timestamptz not null default now(),
    unique (account_id, name)
);

create table if not exists service_types (

    name text not null,
    sort_order integer not null default 0,
    is_active boolean not null default true,
    created_at timestamptz not null default now(),
    unique (account_id, name)
);

create or replace function average_earnings_per_hour_before_shift(
    p_account_id uuid,
    p_user_id uuid,
    p_shift_id bigint
)
returns numeric
language sql
stable
as $$
    select avg(earnings_total / nullif(duration_minutes::numeric / 60, 0))
    from shifts
    where account_id = p_account_id
      and user_id = p_user_id
      and id <> p_shift_id
      and status = 'completed'
      and earnings_total is not null
      and duration_minutes > 0;
$$;

create or replace function get_personal_stats(
    p_account_id uuid,
    p_user_id uuid
)
returns table (
    shifts_count bigint,
    orders_count bigint,
    total_earnings numeric,
    avg_eph numeric
)
language sql
stable
as $$
    select
        count(id) as shifts_count,
        coalesce(sum(orders_count), 0) as orders_count,
        coalesce(sum(earnings_total), 0) as total_earnings,
        coalesce(avg(earnings_total / nullif(duration_minutes::numeric / 60, 0)), 0) as avg_eph
    from shifts
    where account_id = p_account_id
      and user_id = p_user_id
      and status = 'completed';
$$;

create or replace function close_active_shift(
    p_account_id uuid,
    p_telegram_id bigint,
    p_earnings_total numeric
)
returns setof shifts
language plpgsql
as $$
declare
    v_user_id uuid;
    v_shift_id bigint;
    v_now timestamptz := now();
    v_minutes integer;
begin
    select id into v_user_id
    from users
    where telegram_id = p_telegram_id;

    if v_user_id is null then
        return;
    end if;

    select id into v_shift_id
    from shifts
    where account_id = p_account_id
      and user_id = v_user_id
      and status = 'active'
    order by started_at desc
    limit 1
    for update;

    if v_shift_id is null then
        return;
    end if;

    select greatest(1, floor(extract(epoch from (v_now - started_at)) / 60)::integer)
    into v_minutes
    from shifts
    where id = v_shift_id;

    return query
    update shifts
    set ended_at = v_now,
        duration_minutes = v_minutes,
        earnings_total = p_earnings_total,
        status = 'completed',
        updated_at = v_now
    where id = v_shift_id
    returning *;
end;
$$;

create or replace function create_order_for_account_user(
    p_account_id uuid,
    p_telegram_id bigint,
    p_source_mode text,
    p_external_order_id text default null,
    p_currency text default 'RSD',
    p_zone_id uuid default null,
    p_service_type_id uuid default null,
    p_source text default null,
    p_district text default null,
    p_position_district text default null,
    p_pickup_district text default null,
    p_dropoff_district text default null,
    p_platform text default null,
    p_amount numeric default null,
    p_notes text default null
)
returns setof orders
language plpgsql
as $$
declare
    v_user_id uuid;
    v_shift_id bigint;
    v_amount numeric := coalesce(p_amount, 0);
    v_order orders%rowtype;
begin
    select id into v_user_id
    from users
    where telegram_id = p_telegram_id;

    if v_user_id is null then
        raise exception 'User not found';
    end if;

    select id into v_shift_id
    from shifts
    where account_id = p_account_id
      and user_id = v_user_id
      and status = 'active'
    order by started_at desc
    limit 1
    for update;

    insert into orders (
        account_id,
        user_id,
        shift_id,
        external_order_id,
        currency,
        zone_id,
        service_type_id,
        source,
        status,
        ordered_at,
        completed_at,
        notes,
        amount,
        district,
        position_district,
        pickup_district,
        dropoff_district,
        platform,
        source_mode,
        created_at,
        updated_at
    )
    values (
        p_account_id,
        v_user_id,
        v_shift_id,
        p_external_order_id,
        coalesce(p_currency, 'RSD'),
        p_zone_id,
        p_service_type_id,
        p_source,
        'completed',
        now(),
        now(),
        p_notes,
        v_amount,
        coalesce(p_district, p_position_district),
        p_position_district,
        p_pickup_district,
        p_dropoff_district,
        p_platform,
        p_source_mode,
        now(),
        now()
    )
    returning * into v_order;

    if v_shift_id is not null then
        update shifts
        set orders_count = orders_count + 1,
            earnings_total = earnings_total + v_amount,
            updated_at = now()
        where id = v_shift_id;
    end if;

    return next v_order;
end;
$$;

create or replace view v_daily_order_stats as
select
    account_id,
    ordered_at::date as day,
    count(*) as orders_count,
    coalesce(sum(amount), 0::numeric) as earnings_total,
    coalesce(avg(amount), 0::numeric) as avg_order_amount
from orders
group by account_id, ordered_at::date;

create or replace view v_user_daily_stats as
select
    account_id,
    user_id,
    ordered_at::date as day,
    count(*) as orders_count,
    coalesce(sum(amount), 0::numeric) as earnings_total
from orders
group by account_id, user_id, ordered_at::date;

create or replace view v_zone_stats as
select
    account_id,
    zone_id,
    count(*) as orders_count,
    coalesce(sum(amount), 0::numeric) as earnings_total,
    coalesce(avg(amount), 0::numeric) as avg_order_amount
from orders
group by account_id, zone_id;
